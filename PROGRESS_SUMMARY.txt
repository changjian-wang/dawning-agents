🚀 快速开始
═══════════════════════════════════════════════════════════════════════════════

  1. 安装
     dotnet add package Dawning.Agents.Core

  2. 配置 (appsettings.json)
     {
       "LLM": {
         "ProviderType": "Ollama",
         "Model": "qwen2.5:0.5b",
         "Endpoint": "http://localhost:11434"
       }
     }

  3. 使用
     services.AddLLMProvider(configuration);
     services.AddBuiltInTools();
     services.AddReActAgent(options => {
         options.Name = "MyAgent";
         options.Instructions = "你是一个智能助手";
     });

     var agent = sp.GetRequiredService<IAgent>();
     var response = await agent.RunAsync("今天天气怎么样？");
